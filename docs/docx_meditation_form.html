208    def set_rows_height(
209        self, table: Table, rows_height_dataset: Mapping[str, Length]
210    ) -> None:
211        """Apply row heights using index-based rules."""
212        rules = [
213            (lambda i: i == 2 or 4 <= i <= 8, "small_text"),
214            (lambda i: i == 3, "section"),
215            (lambda i: i == 10, "major_section"),
216            (lambda i: i == 16, "tiny_gap"),
217        ]
218
219        for idx, row in enumerate(table.rows):
220            key = self._resolve_height_key(idx, rules, default="base_text")
221            self.set_row_height(row, rows_height_dataset[key])
222
223    def _resolve_height_key(
224        self,
225        idx: int,
226        rules: Iterable,
227        *,
228        default: str,
229    ) -> str:
230        """Resolve height key for a row index."""
231        for predicate, key in rules:
232            if predicate(idx):
233                return key
234        return default

236    def _create_table(self, rows: int, columns: int) -> Table:
237        """Create table with borders."""

239        self.set_table_borders(table, "000000")

242    def set_table_borders(self, table: Table, color: str) -> None:
243        """Apply uniform table borders."""
244        tbl = table._tbl
245        borders = OxmlElement("w:tblBorders")
246
247        for edge in ("top", "left", "bottom", "right", "insideH", "insideV"):
248            e = OxmlElement(f"w:{edge}")
249            e.set(ns.qn("w:val"), "single")
250            e.set(ns.qn("w:sz"), "4")
251            e.set(ns.qn("w:space"), "0")
252            e.set(ns.qn("w:color"), color)
253            borders.append(e)
254
255        tbl.tblPr.append(borders)

267        trPr = tr.get_or_add_trPr()

269        trHeight = OxmlElement("w:trHeight")

271        trHeight.set(qn("w:hRule"), "atLeast")
